# 4-track-glockenspiel-plugin — distributable product (VST3 / AU / Standalone).
# Thin wrapper around dehli_musikk_sampler_engine (dm::ManifestPluginProcessor).

juce_add_plugin(FourTrackGlockenspiel
    COMPANY_NAME            "DehliMusikk"
    BUNDLE_ID               com.dehlimusikk.fourtrackglockenspiel
    VERSION                 0.1.0
    PLUGIN_MANUFACTURER_CODE Dhmk
    PLUGIN_CODE              Gl4t
    FORMATS                 VST3 AU Standalone
    PRODUCT_NAME            "4-track Glockenspiel"
    IS_SYNTH                TRUE
    NEEDS_MIDI_INPUT        TRUE
    NEEDS_MIDI_OUTPUT       FALSE
    IS_MIDI_EFFECT          FALSE
    EDITOR_WANTS_KEYBOARD_FOCUS FALSE
    COPY_PLUGIN_AFTER_BUILD TRUE
    VST3_CATEGORIES         "Instrument" "Sampler"
    AU_MAIN_TYPE            "kAudioUnitType_MusicDevice"
)

target_sources(FourTrackGlockenspiel PRIVATE
    source/Plugin.cpp
)

target_compile_definitions(FourTrackGlockenspiel PUBLIC
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_VST3_CAN_REPLACE_VST2=0
)

target_link_libraries(FourTrackGlockenspiel PRIVATE
    dehli_musikk_sampler_engine
    juce::juce_audio_utils
    juce::juce_recommended_config_flags
    juce::juce_recommended_warning_flags
)

# Embedded asset bundle (generated by ds-plugin-converter into assets/).
# The manifest is embedded SPLIT (manifest/index.json + modes/ + optional partials/)
# and merged at load — NOT the single manifest.json. Mode slugs and partial names
# share one flat resource namespace, so they must not collide.
file(GLOB GLOCK_ASSETS CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/index.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/modes/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/manifest/partials/*.json"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/samples/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/ir/*.flac"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.png"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpg"
    "${CMAKE_CURRENT_SOURCE_DIR}/assets/images/*.jpeg")

if(GLOCK_ASSETS)
    juce_add_binary_data(FourTrackGlockenspielAssets SOURCES ${GLOCK_ASSETS})
    target_link_libraries(FourTrackGlockenspiel PRIVATE FourTrackGlockenspielAssets)
    target_compile_definitions(FourTrackGlockenspiel PRIVATE FOURTRACKGLOCKENSPIEL_HAS_ASSETS=1)
    list(LENGTH GLOCK_ASSETS GLOCK_ASSET_COUNT)
    message(STATUS "FourTrackGlockenspiel: embedding ${GLOCK_ASSET_COUNT} asset file(s) from assets/")
else()
    message(WARNING "FourTrackGlockenspiel: no assets in assets/ — plugin will be silent. "
                    "Generate them with the dmse_convert target first.")
endif()

# Install the disk sample pack (default) into the shared support folder the engine reads.
dmse_setup_plugin(FourTrackGlockenspiel "4-track Glockenspiel")
